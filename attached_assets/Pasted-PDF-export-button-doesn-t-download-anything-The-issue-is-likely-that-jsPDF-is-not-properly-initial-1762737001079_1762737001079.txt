PDF export button doesn't download anything. The issue is likely that jsPDF 
is not properly initialized.

In ExportManager.js, update the exportPDF method to check if jsPDF loaded:

async exportPDF(options = {}) {
  // Check if jsPDF is loaded
  if (typeof window.jspdf === 'undefined' && typeof jsPDF === 'undefined') {
    console.error('jsPDF library not loaded');
    Modal.showError('PDF library not loaded. Please refresh the page.');
    return;
  }

  // Use correct jsPDF reference (it's in window.jspdf.jsPDF)
  const { jsPDF } = window.jspdf || window;

  console.log('Generating PDF...');

  // Get canvas image
  const imgData = this.canvasManager.toDataURL({
    multiplier: 2,
    format: 'png'
  });

  if (!imgData || imgData.length < 100) {
    console.error('Failed to generate canvas image for PDF');
    Modal.showError('Failed to generate PDF');
    return;
  }

  // Rest of the code stays the same...
  const pdf = new jsPDF({
    orientation: options.orientation || 'landscape',
    unit: 'in',
    format: options.format || 'letter'
  });

  // ... continue with existing PDF generation code
  
  const filename = `${this.state.get('metadata.projectName') || 'layout'}.pdf`;
  pdf.save(filename);

  Modal.showSuccess('PDF exported successfully!');
  this.eventBus.emit('export:pdf:complete', filename);
  
  return pdf;
}

Also check that jsPDF is loaded correctly in index.html. The script tag should be:
<script src="https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js"></script>

The info panel (right sidebar) covers the toolbar on small screens.

In layout.css, update the media query:

@media (max-width: 1024px) {
  .info-panel {
    display: none;  /* Hide on tablets/small screens */
  }
}

Alternatively, make the info panel float over the canvas instead of taking up space:

@media (max-width: 1024px) {
  .info-panel {
    position: absolute;
    top: 60px;  /* Below toolbar */
    right: 16px;
    width: 280px;
    max-height: calc(100vh - 80px);
    z-index: 100;
    box-shadow: var(--shadow-xl);
  }
}

This way it floats over the canvas on small screens instead of pushing content.

Add a "+New" button to the toolbar to start a fresh layout.

In index.html, add this button to the first toolbar-group (File Operations):

<div class="toolbar-group">
  <button id="btn-new" class="btn">
    <svg viewBox="0 0 24 24" fill="currentColor">
      <path d="M19,13H13V19H11V13H5V11H11V5H13V11H19V13Z"/>
    </svg>
    New
  </button>
  <button id="btn-save" class="btn btn-icon">
    <!-- existing save button -->
  </button>
</div>

In App.js, add the handler in setupToolbarHandlers():

// New layout
const newBtn = document.getElementById('btn-new');
if (newBtn) {
  newBtn.addEventListener('click', async () => {
    const confirmed = confirm('Start a new layout? Any unsaved changes will be lost.');
    if (confirmed) {
      // Clear everything
      this.state.reset();
      this.canvasManager.clear();
      this.canvasManager.showEmptyState();
      this.renderFloorPlanList();
      this.updateInfoPanel();
      Modal.showSuccess('New layout started');
    }
  });
}

This will clear the canvas and reset to empty state.
